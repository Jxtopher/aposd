# Installation


 apt-get install cmake libpcre3-dev zlib1g-dev libgcrypt11-dev libicu-dev python
 git clone https://github.com/artyom-beilis/cppcms.git cppcms
 cd cppcms
 mkdir build
 cd build
 cmake ..
 make
 make test
 make install
 export LD_LIBRARY_PATH=/usr/local/lib
